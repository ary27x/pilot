PILOT CFG

<SOURCE> -> <STATEMENT>*
<STATEMENT> -> ID is <VALUE> | get <INPUT> | display <EXPRESSION>* |if <CONDITION> : <STATEMENT>* <AFTER_IF> | RETURN INT
<CONDITION> -> <VALUE> <OPERATOR> <VALUE>
<OPERATOR> -> == | != | < | <= | > | >=
<AFTER_IF> -> ELSE <STATEMENT>* : | _EPSILON_
<INPUT> -> <INPUT_PRECEDE> ID
<INPUT_PRECEDE> -> ID , <INPUT_PRECEDE> | _EPSILON_   
<VALUE> -> ID | INT
<EXPRESSION> -> ID | INT | " STRING "
